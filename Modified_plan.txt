PART 3 — DAILY RESET LOGIC
═══════════════════════════════════════════════════════════

Create: services/reset_service.py

Logic:
- Before checking any daily limit, call check_and_reset_daily(user)
- If user.jobs_reset_at < today: set jobs_seen_today=0, jobs_reset_at=today
- If user.cover_letters_reset_at < today: set cover_letters_today=0, 
  cover_letters_reset_at=today
- This runs lazily (on each request) — no scheduled job needed

async def check_and_reset_daily(telegram_id: int, db_pool):
    today = date.today()
    await db_pool.execute("""
        UPDATE users SET
          jobs_seen_today = CASE 
            WHEN jobs_reset_at < $2 THEN 0 
            ELSE jobs_seen_today END,
          jobs_reset_at = CASE 
            WHEN jobs_reset_at < $2 THEN $2 
            ELSE jobs_reset_at END,
          cover_letters_today = CASE 
            WHEN cover_letters_reset_at < $2 THEN 0 
            ELSE cover_letters_today END,
          cover_letters_reset_at = CASE 
            WHEN cover_letters_reset_at < $2 THEN $2 
            ELSE cover_letters_reset_at END
        WHERE telegram_id = $1
    """, telegram_id, today)

  def compute_match_score(user_skills: list, job_skills: list) -> int:
      if not job_skills:
          return 50  # unknown, show neutral
      user_set = set(s.lower() for s in user_skills)
      job_set = set(s.lower() for s in job_skills)
      if not job_set:
          return 50
      matched = user_set.intersection(job_set)
      score = int((len(matched) / len(job_set)) * 100)
      return score
  
  Display in job card:
  - Score >= 70%: "🟢 87% match — React ✅ TypeScript ✅ GraphQL ❌"

async def send_due_reminders(bot):
  reminders = await db.fetch("""
    SELECT r.*, a.company_name, a.job_title, a.id as app_id,
           u.telegram_id
    FROM reminders r
    JOIN applications a ON r.application_id = a.id  
    JOIN users u ON r.telegram_id = u.telegram_id
    WHERE r.remind_at <= NOW() 
      AND r.sent = FALSE
      AND r.reminder_type = 'followup'
    LIMIT 50
  """)
  
  for reminder in reminders:
    days_ago = (datetime.now() - reminder.applied_at).days
    
    await bot.send_message(
      chat_id=reminder.telegram_id,
      text=f"👋 It's been {days_ago} days since you applied to\n"
           f"*{reminder.company_name} — {reminder.job_title}*\n\n"
           f"Any update?",
      reply_markup=InlineKeyboardMarkup([[
        InlineKeyboardButton("✅ Got Interview!", callback_data=f"reminder_interview_{reminder.app_id}"),
        InlineKeyboardButton("❌ Rejected", callback_data=f"reminder_rejected_{reminder.app_id}"),
      ],[
        InlineKeyboardButton("🔇 Still Waiting", callback_data=f"reminder_waiting_{reminder.app_id}"),
      ]])
    )
    
    -- Mark reminder as sent
    await db.execute("UPDATE reminders SET sent=TRUE WHERE id=$1", reminder.id)
    -- Update application last_reminded_at
    await db.execute("UPDATE applications SET last_reminded_at=NOW(), reminder_count=reminder_count+1 WHERE id=$1", reminder.app_id)
    
    -- Avoid Telegram rate limit
    await asyncio.sleep(0.1)
